def index
  @data_exports = DataExport.all.order(created_at: :desc)
  @params = {}
  if params[:export_type]
    @params = params.require(:data_export).permit(:export_type).reject { |_, v| v.blank? }
    @data_exports = @data_exports.where(@params)
  end
end
