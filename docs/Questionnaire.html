def verbal_status
  if acc_status == "rsvp_denied"
    "Not Attending"
  elsif acc_status == "rsvp_confirmed"
    "Accepted & Attending"
  elsif acc_status == "accepted"
    "Accepted, Awaiting RSVP"
  elsif acc_status == "pending"
    "Pending Review"
  elsif ["waitlist", "late_waitlist"].include? acc_status
    "Waitlisted"
  elsif acc_status == "denied"
    "Denied"
  end
end
